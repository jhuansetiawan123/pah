.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷* .*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*.*࠷*
